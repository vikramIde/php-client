  6  * Simple Logging Manager.
  7  * This does an error_log for now
  8  * Potential frameworks to use are PEAR logger, log4php from Apache
  9  */
 10 class BlockCypherLoggingManager
 11 {
 12     /**
 13      * Default Logging Level
 14      */
 15     const DEFAULT_LOGGING_LEVEL = 0;
 16 
 17     /**
 18      * Logger Name
 19      * @var string
 20      */
 21     private $loggerName;
 22 
 23     /**
 24      * Log Enabled
 25      *
 26      * @var bool
 27      */
 28     private $isLoggingEnabled;
 29 
 30     /**
 31      * Configured Logging Level
 32      *
 33      * @var int|mixed

 44     /**
 45      * Default Constructor
 46      */
 47     public function __construct()
 48     {
 49         // To suppress the warning during the date() invocation in logs, we would default the timezone to GMT.
 50         if (!ini_get('date.timezone')) {
 51             date_default_timezone_set('GMT');
 52         }
 53 
 54         $config = BlockCypherConfigManager::getInstance()->getConfigHashmap();
 55 
 56         $this->isLoggingEnabled = (array_key_exists('log.LogEnabled', $config) && $config['log.LogEnabled'] == '1');
 57 
 58         if ($this->isLoggingEnabled) {
 59             $this->loggerFile = ($config['log.FileName']) ? $config['log.FileName'] : ini_get('error_log');
 60             $loggingLevel = strtoupper($config['log.LogLevel']);
 61             $this->loggingLevel =
 62                 (isset($loggingLevel) && defined(__NAMESPACE__ . "\\BlockCypherLoggingLevel::$loggingLevel")) ?
 63                     constant(__NAMESPACE__ . "\\BlockCypherLoggingLevel::$loggingLevel") :
 64                     BlockCypherLoggingManager::DEFAULT_LOGGING_LEVEL;
 65         }
 66     }
 67 
 68     /**
 69      * Returns the singleton object
 70      *
 71      * @param string $loggerName
 72      * @return $this
 73      */
 74     public static function getInstance($loggerName = __CLASS__)
 75     {
 76         $instance = new self();
 77         $instance->setLoggerName($loggerName);
 78         return $instance;
 79     }

101     /**
102      * Default Logger
103      *
104      * @param string $message
105      * @param int $level
106      */
107     private function log($message, $level = BlockCypherLoggingLevel::INFO)
108     {
109         if ($this->isLoggingEnabled) {
110             $config = BlockCypherConfigManager::getInstance()->getConfigHashmap();
111 
112             if (isset($config['mode'])) {
113                 $configMode = $config['mode'];
114             } else {
115                 // mode has not been configured by user
116                 $configMode = null;
117             }
118 
119             // Check if logging in live
120             if ($configMode == 'live') {
121                 // Live should not have logging level above INFO.
122                 if ($this->loggingLevel >= BlockCypherLoggingLevel::INFO) {
123                     // If it is at Debug Level, throw an warning in the log.
124                     if ($this->loggingLevel == BlockCypherLoggingLevel::DEBUG) {
125                         error_log("[" . date('d-m-Y h:i:s') . "] " . $this->loggerName . ": ERROR\t: Not allowed to keep 'Debug' level for Live Environments. Reduced to 'INFO'\n", 3, $this->loggerFile);
126                     }
127                     // Reducing it to info level
128                     $this->loggingLevel = BlockCypherLoggingLevel::INFO;
129                 }
130             }
131 
132             if ($level <= $this->loggingLevel) {
133                 error_log("[" . date('d-m-Y h:i:s') . "] " . $this->loggerName . ": $message\n", 3, $this->loggerFile);
134             }
135         }
136     }
